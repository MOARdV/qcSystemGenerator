150 bool dustAvailable(const Protoplanet& protoplanet) const
151 {
152 if (protoplanet.sma <= 0.0)
153 {
154 return false;
155 }
156
157 const double innerLimit = innerEffectLimit(protoplanet.sma, protoplanet.eccentricity, protoplanet.mass);
158 const double outerLimit = outerEffectLimit(protoplanet.sma, protoplanet.eccentricity, protoplanet.mass);
159 bool available = false;
160 auto band = availableDust.begin();
161
162 while (band != availableDust.end() && band->outerEdge < innerLimit)
163 {
164 ++band;
165 }
166
167 if (band != availableDust.end())
168 {
169 available = band->dustPresent;
170 }
171
172 while (band != availableDust.end() && band->innerEdge < outerLimit)
173 {
174 available = available || band->dustPresent;
175 ++band;
176 }
177
178 return available;
179 }
